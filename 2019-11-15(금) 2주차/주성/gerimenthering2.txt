void dfs(int dy,int dx)
{
  //printf("%d %d\n",y,x);
  for(int d2=1;d2<=n;d2++)
  {
    for(int d1=1;d1<=n;d1++)
    {
      int y=dy;
      int x=dx;
      
      if(dx>= dx+d1+d2) continue;
      if(dx+d1+d2 >n) continue;
      if(dy>dy-d1) continue;
      if(n<dy+d2) continue;
      make_nemo(y,x,d1,d2);
      //printf("%d %d %d %d\n",y,x,d1,d2);

      
    }
  }
}
void go()
{
  for(int i=1;i<n-1;i++)
  {
    for(int j=2;j<=n-1;j++)
    {
      dfs(i,j);
    }
  }
}
int main() 
{
  scanf("%d",&n);
  for(int i=1;i<=n;i++)

    for(int j=1;j<=n;j++)

      scanf("%d",&map[i][j]);
    }
  }
  
  go();

}
